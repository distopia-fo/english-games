            { base: "eat", past: "ate", russian: "–µ—Å—Ç—å, –∫—É—à–∞—Ç—å" },

            { base: "take", past: "took", russian: "–±—Ä–∞—Ç—å" },
            { base: "buy", past: "bought", russian: "–ø–æ–∫—É–ø–∞—Ç—å" },
            { base: "write", past: "wrote", russian: "–ø–∏—Å–∞—Ç—å" },
            { base: "speak", past: "spoke", russian: "–≥–æ–≤–æ—Ä–∏—Ç—å" },
            { base: "read", past: "read", russian: "—á–∏—Ç–∞—Ç—å" }
        ];
        
        let currentIndex = 0;
        let correctCount = 0;
        let answered = false;
        
        const infinitiveEl = document.getElementById('infinitive');
        const russianEl = document.getElementById('russian');
        const answerInput = document.getElementById('answer-input');
        const checkBtn = document.getElementById('check-btn');
        const nextBtn = document.getElementById('next-btn');
        const hintBtn = document.getElementById('hint-btn');
        const feedbackEl = document.getElementById('feedback');
        const progressText = document.getElementById('progress-text');
        const scoreText = document.getElementById('score-text');
        const verbsList = document.getElementById('verbs-list');
        
        // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º —Å–ø–∏—Å–æ–∫ –≤—Å–µ—Ö –≥–ª–∞–≥–æ–ª–æ–≤
        function renderVerbsList() {
            verbsList.innerHTML = VERBS.map(v => 
                `<div class="verb-badge">
                    <strong>${v.base}</strong>
                    ${v.past}
                    <div style="font-size: 0.8rem; color: #6b7280;">${v.russian}</div>
                </div>`
            ).join('');
        }
        
        function updateDisplay() {
            const verb = VERBS[currentIndex];
            infinitiveEl.textContent = verb.base;
            russianEl.textContent = verb.russian;
            progressText.textContent = `–ì–ª–∞–≥–æ–ª ${currentIndex + 1} –∏–∑ ${VERBS.length}`;
            scoreText.textContent = `‚úÖ ${correctCount} / ${VERBS.length}`;
            answerInput.value = '';
            answerInput.focus();
            answered = false;
            feedbackEl.classList.remove('show', 'correct', 'wrong');
        }
        
        function checkAnswer() {
            if (answered) return;
            
            const userAnswer = answerInput.value.trim().toLowerCase();
            const correctAnswer = VERBS[currentIndex].past.toLowerCase();
            
            if (userAnswer === '') {
                feedbackEl.textContent = '‚úèÔ∏è –í–≤–µ–¥–∏—Ç–µ –æ—Ç–≤–µ—Ç!';
                feedbackEl.className = 'feedback show wrong';
                return;
            }
            
            if (userAnswer === correctAnswer) {
                feedbackEl.textContent = '‚úÖ –ü—Ä–∞–≤–∏–ª—å–Ω–æ! –¢–∞–∫ –¥–µ—Ä–∂–∞—Ç—å!';
                feedbackEl.className = 'feedback show correct';
                if (!answered) {
                    correctCount++;
                    scoreText.textContent = `‚úÖ ${correctCount} / ${VERBS.length}`;
                }
                answered = true;
            } else {
                feedbackEl.textContent = `‚ùå –ù–µ–ø—Ä–∞–≤–∏–ª—å–Ω–æ. –ü–æ–ø—Ä–æ–±—É–π –µ—â—ë —Ä–∞–∑!`;
                feedbackEl.className = 'feedback show wrong';
            }
        }
        
        function nextVerb() {
            if (currentIndex < VERBS.length - 1) {
                currentIndex++;
                updateDisplay();
            } else {
                if (correctCount === VERBS.length) {
                    feedbackEl.textContent = 'üèÜ –ü–û–ó–î–†–ê–í–õ–Ø–Æ! –¢—ã –≤—ã—É—á–∏–ª –≤—Å–µ –≥–ª–∞–≥–æ–ª—ã!';
                } else {
                    feedbackEl.textContent = 'üéØ –¢—ã –ø—Ä–æ—à—ë–ª –≤—Å–µ –≥–ª–∞–≥–æ–ª—ã! –ù–∞—á–Ω–∏ –∑–∞–Ω–æ–≤–æ –∏ –ø–æ–ø—Ä–æ–±—É–π –µ—â—ë!';
                }
                feedbackEl.className = 'feedback show correct';
            }
        }
        
        function showHint() {
            const verb = VERBS[currentIndex];
            const hint = verb.past;
            const maskedHint = hint[0] + '_'.repeat(hint.length - 1);
            feedbackEl.textContent = `üí° –ü–æ–¥—Å–∫–∞–∑–∫–∞: —Å–ª–æ–≤–æ –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è –Ω–∞ "${hint[0]}", –≤—Å–µ–≥–æ –±—É–∫–≤: ${hint.length}`;
            feedbackEl.className = 'feedback show';
        }
        
        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
        renderVerbsList();
        updateDisplay();
        
        // –û–±—Ä–∞–±–æ—Ç—á–∏–∫–∏
        checkBtn.addEventListener('click', checkAnswer);
        nextBtn.addEventListener('click', nextVerb);
        hintBtn.addEventListener('click', showHint);
        
        answerInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                checkAnswer();
            }
        });
